3   import hr.fer.zemris.java.gui.calc.containers.NumberContainer;
4   import hr.fer.zemris.java.gui.calc.containers.OperatorContainer;
5   
6   /**
7    * Class used for calculation of expressions given by {@link Calculator}.
8    * @author Vilim StaroveÅ¡ki
9    *
10   */
11  public class Calculation {
12  
13  	/**
14  	 * First operand in calculation.
15  	 */
16  	private double firstOperand;
17  	
18  	/**
19  	 * Second operand in calculation.
20  	 */
21  	private double secondOperand;
22  	
23  	/**
24  	 * Type of operation, expected enumeration {@link Operation}
25  	 */
26  	private Operation operationType;
27  	
28  	/**
29  	 * Flag, for using inverse functions or not.
30  	 */
31  	private boolean inverse;
32  	
33  	/**
34  	 * Creates new {@link Calculation} with 2 operands and operation type. Typically used for
35  	 * simple calculation such as + - * / 
36  	 * @param firstArg  First operand in calculation.
37  	 * @param secondArg Second operand in calculation.
38  	 * @param oper type of operation defined within {@link OperatorContainer}.
39  	 */
40  	public Calculation(NumberContainer firstArg, NumberContainer secondArg, OperatorContainer oper) {
41  		
42  		this.firstOperand = Double.parseDouble(firstArg.getValue());
43  		this.secondOperand = Double.parseDouble(secondArg.getValue());
44  		this.operationType = oper.getSymbol();
45  	}
46  	
47  	/**
48  	 * Creates new {@link Calculation} with 1 operand and operation type. Used for functions such as
49  	 * sin, cos, tan, ctg, ln, log, n-th power, 1/x
50  	 * @param firstArg First operand in calculation.
51  	 * @param operationType Type of operation, expected enumeration {@link Operation}
52  	 * @param inverseFunction flag for using inverse functions or no.
53  	 */
54  	public Calculation(NumberContainer firstArg, Operation operationType, boolean inverseFunction) {
55  		
56  		this.firstOperand = Double.parseDouble(firstArg.getValue());
57  		this.inverse = inverseFunction;
58  		this.operationType = operationType;
59  	}
60  	
61  	/**
62  	 * Returns result of this {@link Calculation}.
63  	 * @return result of this {@link Calculation}.
64  	 */
65  	@SuppressWarnings("incomplete-switch")
66  	public NumberContainer getResult() {
67  		
68  		switch(operationType) {
69  		case PLUS:			return new NumberContainer(String.valueOf(firstOperand+secondOperand));
70  		case MINUS:			return new NumberContainer(String.valueOf(firstOperand-secondOperand));
71  		case TIMES:			return new NumberContainer(String.valueOf(firstOperand*secondOperand));
72  		case DIVIDE:		return new NumberContainer(String.valueOf(firstOperand/secondOperand));
73  		case RECIPROCAL:	return new NumberContainer(String.valueOf(1/firstOperand));
74  		}
75  		
76  		if(!inverse) {
77  			switch(operationType) {
78  			case SIN:				return new NumberContainer(String.valueOf(Math.sin(firstOperand)));
79  			case COS:				return new NumberContainer(String.valueOf(Math.cos(firstOperand)));
80  			case TAN:				return new NumberContainer(String.valueOf(Math.tan(firstOperand)));
81  			case CTG:				return new NumberContainer(String.valueOf(1/Math.tan(firstOperand)));
82  			case LOG:				return new NumberContainer(String.valueOf(Math.log10(firstOperand)));
83  			case LN:				return new NumberContainer(String.valueOf(Math.log(firstOperand)));
84  			case NPOWER:			return new NumberContainer(String.valueOf(Math.pow(firstOperand, secondOperand)));
85  			default:				return null;
86  			}
87  		}
88  		else {
89  			switch(operationType) {
90  			case SIN:				return new NumberContainer(String.valueOf(Math.asin(firstOperand)));
91  			case COS:				return new NumberContainer(String.valueOf(Math.acos(firstOperand)));
92  			case TAN:				return new NumberContainer(String.valueOf(Math.atan(firstOperand)));
93  			case CTG:				return new NumberContainer(String.valueOf(1/Math.atan(firstOperand)));
94  			case LOG:				return new NumberContainer(String.valueOf(Math.pow(10, firstOperand)));
95  			case LN:				return new NumberContainer(String.valueOf(Math.pow(Math.E, firstOperand)));
96  			case NPOWER:			return new NumberContainer(String.valueOf(Math.pow(firstOperand, -secondOperand)));
97  			default:				return null;
98  			}
99  		}
100 	}
